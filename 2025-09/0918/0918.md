## ✍ 접근 방식

```ts
const combination = (list, count) => {
    const result = [];
    
    const helper = (arr, rest) => {
        if(arr.length === count) {
            result.push(arr.sort((a, b) => a > b ? 1 : -1).join(""));
            return;
        }
        
        rest.forEach((char, index) => {
            const newRest = rest.slice(index + 1);
            helper([...arr, char], newRest);
        })
    }
    
    helper([], list);
    
    return result;
}
```
